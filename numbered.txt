[367]def format_specs_to_json_table(specs_str: str) -> str:
[368]    """Chuyển đổi chuỗi JSON thông số thành bảng Markdown"""
[369]    if not specs_str or specs_str.strip() in ["", "{}", "None"]:
[370]        return "*(Đang cập nhật)*"
[371]        
[372]    try:
[373]        # Cố gắng đọc chuỗi thành Dictionary
[374]        data = json.loads(specs_str)
[375]        if not data:
[376]            return "*(Đang cập nhật)*"
[377]
[378]        # Kiểm tra xem có phải JSON lồng nhau (Multiple Models) không
[379]        # Ví dụ: {"INVICTA 430": {"Speed": "220"}, "INVICTA 630": {"Speed": "200"}}
[380]        is_nested = any(isinstance(v, dict) for v in data.values())
[381]
[382]        if is_nested:
[383]            # Lấy danh sách các model và tất cả các thông số
[384]            models = [str(k) for k in data.keys() if str(k).strip() != ""]
[385]            if not models: return "*(Đang cập nhật)*"
[386]            
[387]            all_params = []
[388]            for model in models:
[389]                if isinstance(data[model], dict):
[390]                    for k in data[model].keys():
[391]                        if k not in all_params:
[392]                            all_params.append(k)
[393]            
[394]            # Khởi tạo Header của bảng
[395]            header = "| Thông số | " + " | ".join(models) + " |"
[396]            separator = "|---| " + " | ".join(["---" for _ in models]) + " |"
[397]            
[398]            # Điền dữ liệu vào từng hàng
[399]            rows = []
[400]            for param in all_params:
[401]                row = f"| **{param}** |"
[402]                for model in models:
[403]                    val = data[model].get(param, "-") if isinstance(data[model], dict) else "-"
[404]                    row += f" {val} |"
[405]                rows.append(row)
[406]                
[408]            
[409]        else:
[410]            # JSON phẳng (Single Model)
[411]            # Ví dụ: {"Tốc độ": "220", "Khổ giấy": "A4"}
[412]            header = "| Thông số | Giá trị |"
[413]            separator = "|---|---|"
[414]            rows = [f"| **{k}** | {v} |" for k, v in data.items()]
[416]
[417]    except json.JSONDecodeError:
[418]        # Fallback: Nếu lỗi parse JSON, cắt gọn và làm sạch chuỗi thô
